3    import java.io.File;
4    import java.io.IOException;
5    
6    /**
7     7 * This interface is common to both the Neural Net and LUT interfaces.
8     8 * The idea is that you should be able to easily switch the LUT
9     9 * for the Neural Net since the interfaces are identical.
10    10 * @date 20 June 2012
11    11 * @author sarbjit
12    12 *
13    13 */
14   
15   public interface CommonInterface {
16       /**
17        * * @param X The input vector. An array of doubles.
18        * @return The value returned by th LUT or NN for this input vector
19        */
20       public double outputFor(double [] X);
21   
22       /**
23        * This method will tell the NN or the LUT the output
24        * value that should be mapped to the given input vector. I.e.
25        * the desired correct output value for an input.
26        * @param X The input vector
27        * @param argValue The new value to learn
28        * @return The error in the output for that input vector
29        */
30   
31       public double train(double [] X, double argValue);
32   
33       /**
34       * A method to write either a LUT or weights of an neural net to a file.
35       * @param argFile of type File.
36       */
37   
38       public void save(File argFile);
39   
40       /**
41        * Loads the LUT or neural net weights from file. The load must of course
42        * have knowledge of how the data was written out by the save method.
43        * You should raise an error in the case that an attempt is being
44        * made to load data into an LUT or neural net whose structure does not match
45        * the data in the file. (e.g. wrong number of hidden neurons).
46        * @throws IOException
47        */
48       public void load(String argFileName) throws IOException;
49   
50   }
